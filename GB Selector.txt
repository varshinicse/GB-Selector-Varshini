/* ----------------------- SELECT GEARBOX (UPDATED) ----------------------- */
function selectGearbox(seriesVal, nominalTorque, maxTorque, stageIndex, stageRatio){
    const seriesNum = parseInt(seriesVal.replace("s",""));
    let filteredGearboxes = gearboxes.filter(g => g.series === seriesNum);

    // ----- NEW: Restrict first stage gearboxes by ratio -----
    if(stageIndex === 0 && seriesVal === "s1"){ // first stage
        if(stageRatio === 3.75 || stageRatio === 4.25 || stageRatio === 4.5 || stageRatio === 4.71 || stageRatio === 5.05){
            filteredGearboxes = filteredGearboxes.filter(g => g.size.startsWith("L"));
        } else if(stageRatio === 5.67 ||stageRatio === 5.71 || stageRatio === 6.25 || stageRatio === 6.68 || stageRatio === 7.2 || stageRatio === 7.58 || stageRatio === 7.6){
            filteredGearboxes = filteredGearboxes.filter(g => g.size.startsWith("M"));
        } else if(stageRatio === 8.04 || stageRatio === 8.65 || stageRatio === 8.74 || stageRatio === 9.4 || stageRatio === 9.43 || stageRatio === 10.125 || stageRatio === 10.26){
            filteredGearboxes = filteredGearboxes.filter(g => g.size.startsWith("H"));
        }
    }

    // ----- Existing selection logic -----
    let selected = filteredGearboxes
        .filter(g => g.nominal >= nominalTorque && g.rated >= maxTorque)
        .sort((a,b) => a.nominal - b.nominal);
    if(selected.length>0) return selected[0];

    selected = filteredGearboxes
        .filter(g => g.rated >= maxTorque)
        .sort((a,b) => a.rated - b.rated);
    if(selected.length>0) return selected[0];

    if(filteredGearboxes.length>0)
        return filteredGearboxes.sort((a,b)=>b.nominal-a.nominal)[0];

    return gearboxes[gearboxes.length-1];
}

/* ----------------------- CALCULATE RATIOS ----------------------- */
function calculateRatios(){
    const reqRatio=+document.getElementById("totalRatio").value;
    const stages=+document.getElementById("stages").value;
    const power=+document.getElementById("powerKW").value;
    const rpm=+document.getElementById("inputRPM").value;
    const sf=+document.getElementById("serviceFactor").value;

    if(!reqRatio||!stages||!power||!rpm||!sf){
        alert("Fill all fields");
        return;
    }

    let seriesArr=[];
    for(let i=1;i<=stages;i++) seriesArr.push(seriesData[document.getElementById(`stageSeries${i}`).value]);

    let results=[];
    function gen(stage,ratios,total){
        if(stage===stages){
            const deviation=((total-reqRatio)/reqRatio)*100;
            const nominal=(power*60000*total*Math.pow(0.97,stages))/(2*Math.PI*rpm);
            const max=nominal*sf;
            results.push({ratios,total,deviation,nominal,max});
            return;
        }
        seriesArr[stage].forEach(r=>gen(stage+1,[...ratios,r],total*r));
    }
    gen(0,[],1);

    results.sort((a,b)=>Math.abs(a.deviation)-Math.abs(b.deviation));
    let n = +document.getElementById("numOptions").value || 5; // get user input, fallback to 5
    if(n > 10) n = 10; // limit maximum to 10
    top5Results = results.slice(0, n);

    let html=`<div class="card"><h3>Top ${n} Options</h3><table>
        <tr><th>Opt</th><th>Ratios</th><th>Total</th><th>Dev %</th>
        <th>Nom Torque</th><th>Max Torque</th><th>Gearbox</th></tr>`;

    top5Results.forEach((r,i)=>{
        const lastStageSeriesVal=document.getElementById(`stageSeries${stages}`).value;
        const gb=selectGearbox(
            lastStageSeriesVal,
            r.nominal,
            r.max,
            stages-1,
            r.ratios[stages-1]
        );
        const nominalExceeded=r.nominal>gb.nominal;
        const ratedExceeded=r.max>gb.rated;
        const devClass=Math.abs(r.deviation)>3?"high-dev":"";

        html+=`<tr ondblclick="showStageDetails(${i})">
            <td>${i+1}</td>
            <td>${r.ratios.map(x=>x.toFixed(2)).join(" × ")}</td>
            <td>${r.total.toFixed(2)}</td>
            <td class="${devClass}">${r.deviation.toFixed(2)}</td>
            <td>${r.nominal.toFixed(0)}</td>
            <td>${r.max.toFixed(0)}</td>
            <td>${gb.size} ${nominalExceeded||ratedExceeded ? '<span class="bad"> ⚠ Exceeds</span>' : '<span class="ok"> ✓ OK</span>'}</td>
        </tr>`;
    });

    html+=`</table></div>`;
    document.getElementById("results").innerHTML=html;
}

/* ---------------- Stage Details with Safety / SF ---------------- */
function showStageDetails(index){
    const r=top5Results[index];
    if(!r) return;
    let speed=+document.getElementById("inputRPM").value;
    let torque=(+document.getElementById("powerKW").value*60000)/(2*Math.PI*speed);
    let sf=+document.getElementById("serviceFactor").value;

    let html=`<h3>Stage Details – Option ${index+1}</h3><table>
        <tr><th>Stage</th><th>Ratio</th><th>Speed</th>
        <th>Nom Torque</th><th>Max Torque</th><th>Gearbox</th>
        <th>GB Nominal</th><th>GB Rated</th><th>Safety / SF</th></tr>`;

    r.ratios.forEach((ratio,idx)=>{
        const seriesVal=document.getElementById(`stageSeries${idx+1}`).value;
        speed/=ratio;
        torque*=ratio*0.97;
        const maxTorque=torque*sf;
        const gb=selectGearbox(seriesVal, torque, maxTorque, idx, ratio);
        let safety=0;
        if(gb.nominal/torque <= gb.rated/maxTorque){
            safety=(gb.nominal/torque).toFixed(2);
        } else {
            safety=(gb.rated/maxTorque).toFixed(2);
        }
        let safetyClass = safety < 1 ? "bad" : "ok";

        html+=`<tr>
            <td>${idx+1}</td>
            <td>${ratio.toFixed(2)}</td>
            <td>${speed.toFixed(1)}</td>
            <td>${torque.toFixed(1)}</td>
            <td>${maxTorque.toFixed(1)}</td>
            <td>${gb.size}</td>
            <td>${gb.nominal}</td>
            <td>${gb.rated}</td>
            <td class="${safetyClass}">${safety}</td>
        </tr>`;
    });

    html+=`</table>`;
    document.getElementById("modalContent").innerHTML=html;
    document.getElementById("modalOverlay").style.display="flex";
}
